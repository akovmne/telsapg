    function getWeekType() {

      const now = new Date();

      const referenceDate = new Date(2026, 4, 3); // May 3, 2026 = Afternoon week

      const diffTime = now - referenceDate;

      const diffDays = Math.floor(diffTime / (1000 * 60 * 60 * 24));

      const weekNumber = Math.floor(diffDays / 7);

      return weekNumber % 2 === 0 ? "afternoon" : "morning";

    }


    function isAvailable() {

      const now = new Date();

      const day = now.getDay(); // 0=Sunday, 1=Monday

      if (day === 1) return false; // Monday off


      const weekType = getWeekType();

      const hours = now.getHours() + now.getMinutes()/60;


      if (weekType === "morning") {

        return hours >= 7 && hours <= 15;

      } else {

        return hours >= 15 && hours <= 23;

      }

    }


    const statusDiv = document.getElementById('status');

    const shiftDiv = document.getElementById('shift-info');

    const weekType = getWeekType();


    if (isAvailable()) {

      statusDiv.className = "status available";

      statusDiv.textContent = "✅ AVAILABLE RIGHT NOW";

    } else {

      statusDiv.textContent = "❌ Not available at the moment";
